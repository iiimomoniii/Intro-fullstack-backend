//install sequelize
npm i -g sequelize-cli

//initial sequelize for project
sequelize init

//generate a model example Products
sequelize model:generate --name Products --attributes "name:string, image:string, stock:integer, price:integer" --underscored true

//generate seed for mockup data example Products
sequelize seed:generate --name seed-Products


//generate database
sequelize db:migrate

//create table and insert data from seed to database
sequelize db:seed:all

//use database from sqlite
$env:NODE_ENV="development"
nodemon server.js
